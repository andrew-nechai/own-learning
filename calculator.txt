// инкремент (++) и декремент (--).
#include <iostream>   // подключение библиотеки работы с консолью
 
using namespace std;  // пространство имен

int main()
{
	setlocale(LC_ALL, "ru");
	float a, b;
	int c;
	cout << "Введите два числа ";
	if (cin >> a >> b)
	{
		cout << "Выберите операцию:" << endl << "1.Умножение - 1" << endl << "2.Деление - 2" << endl << "3.Сложение - 3" << endl << "4.Вычитание - 4" << endl;
		cin >> c;
		switch (c)
		{
		case 1:
			cout << "Результат умножения - (" << a*b << ")";
			break;
		case 2:
			if (b != 0) {
				cout << "Результат деления - (" << (float)(a / b) << ")";
				break;
			}
			else
			{
				cout << "На ноль делить нельзя!";
				break;
			}
		case 3:
			cout << "Результат сложения - (" << (a + b) << ")";
			break;
		case 4:
			cout << "Результат вычитания - (" << (a / b) << ")";
			break;
		default:
			cout << "Выбери правильный вариант!";
			break;
		}
	}
	else
	{
		cout << "Вы ввели букву!";
	}
}
